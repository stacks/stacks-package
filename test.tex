% include a tag from the Stacks project
\newcommand\stacksinclude[1]{%
  % download the tag's TeX code
  \download[#1.tex]{http://stacks.math.columbia.edu/data/tag/#1/content/statement/raw}%
  % immediately typeset it (we could do something else too)
  \input{#1}%
}
